/* ---------------- CLICK GAME ---------------- */ let clicks=0; let time=10; let timer=null; let started=false; let leaderboard=[]; function startGame(){ let name=document.getElementById("playerName").value; if(!name){ alert("Enter name!"); return; } clicks=0; time=10; started=true; document.getElementById("score").innerText=0; document.getElementById("time").innerText=10; document.getElementById("msg").innerHTML="🔥 GO!"; clearInterval(timer); timer=setInterval(()=>{ time--; document.getElementById("time").innerText=time; if(time<=0){ endGame(name); } },1000); } function clickMe(){ if(!started){ document.getElementById("msg").innerHTML="Press START first!"; return; } clicks++; document.getElementById("score").innerText=clicks; } function endGame(name){ started=false; clearInterval(timer); leaderboard.push({name:name,score:clicks}); leaderboard.sort((a,b)=>b.score-a.score); renderBoard(); document.getElementById("msg").innerHTML="💀 TIME UP!"; } function renderBoard(){ let html=""; leaderboard.slice(0,5).forEach((p,i)=>{ html+=`${i+1}. ${p.name} → ${p.score}
`; }); document.getElementById("board").innerHTML=html; } /* RESET */ function resetGame(){ clicks=0; time=10; started=false; clearInterval(timer); document.getElementById("score").innerText=0; document.getElementById("time").innerText=10; document.getElementById("msg").innerHTML=""; }
